prototypeObject = {
fullName: function(){
return this.firstName + " " + this.lastName
}
}
var person = Object.create(prototypeObject, {
'firstName': {
value: "Virat",
writable: true,
enumerable: true
},
'lastName': {
value: "Kohli",
writable: true,
enumerable: true
}
})
console.log(person) // Object with prototype object as prototypeObject and properties as firstName and lastName
